```gdscript
class_name CN_NetSync
extends Component
# No constructor args; instantiate with CN_NetSync.new()
# Priority is declared via @export_group sentinels on sibling components:
#   "REALTIME" -> ~60 Hz unreliable
#   "HIGH"     -> 20 Hz unreliable
#   "MEDIUM"   -> 10 Hz reliable
#   "LOW"      -> 1 Hz reliable
#   "SPAWN_ONLY" -> sent at spawn only, never continuous
#   "LOCAL"      -> never synced
```
