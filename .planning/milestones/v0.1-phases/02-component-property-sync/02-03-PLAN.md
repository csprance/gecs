    _dispatch_batch() RPC dispatch (ported from sync_property_handler.gd lines 329-343):
    ```gdscript
    func _dispatch_batch(priority: int) -> void:
        if _pending[priority].is_empty():
            return
        var batch = _pending[priority].duplicate(true)
        _pending[priority].clear()

        var is_unreliable = priority <= CN_NetSync.Priority.HIGH  # REALTIME or HIGH
        if _ns.net_adapter.is_server():
            if is_unreliable:
                _ns._sync_components_unreliable.rpc(batch)
            else:
                _ns._sync_components_reliable.rpc(batch)
        else:
            if is_unreliable:
                _ns._sync_components_unreliable.rpc_id(1, batch)
            else:
                _ns._sync_components_reliable.rpc_id(1, batch)
    ```

    Note: In tests, MockNetworkSync does NOT have real .rpc() callables — it has plain methods that append to arrays. The tests verify rpc_calls arrays, not actual network sends.     # Set in _init: _unreliable_callable = _ns._sync_components_unreliable
    # Call: _unreliable_callable.rpc(batch)

Since both cases call the method the same way in GDScript via `_ns._sync_components_unreliable(batch)` for direct call or `_ns._sync_components_unreliable.rpc(batch)` for RPC, structure tests to capture the non-rpc call path, and production code adds .rpc() suffix.     func _apply_component_data(entity: Entity, comp_data: Dictionary) -> void:
        var net_sync = entity.get_component(CN_NetSync)
        _ns._applying_network_data = true
        for comp_type in comp_data.keys():
            var comp = _find_component_by_type(entity, comp_type)
            if comp == null:
                continue
            var props = comp_data[comp_type]
            for prop in props.keys():
                var value = props[prop]
                if prop in comp:
                    comp.set(prop, value)
                    # Update CN_NetSync cache silently to prevent re-detection
                    if net_sync:
                        net_sync.update_cache_silent(comp, prop, value)
        _ns._applying_network_data = false
