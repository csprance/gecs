Implement structural relationship query resolution across three files. The approach: classify relationships into structural (archetype-level) vs post-filter (entity-level), integrate structural relationships into cache keys and archetype matching.

### query_builder.gd changes:

1. **Add classification arrays** (alongside existing `_relationships` / `_exclude_relationships`):

```gdscript
var _structural_rel_keys: Array = []       # Exact rel:// slot keys for archetype matching
var _wildcard_rel_types: Array = []        # Relation paths for wildcard index lookup
var _post_filter_relationships: Array = [] # Property-query and script-target rels (entity-level)
var _structural_ex_rel_keys: Array = []
var _wildcard_ex_rel_types: Array = []
var _post_filter_ex_relationships: Array = []
```

2. **Modify `with_relationship()`**: After storing `_relationships`, classify each relationship:
   - `rel._is_query_relationship` → `_post_filter_relationships` (property queries can't be structural)
   - `rel.target is Script` → `_wildcard_rel_types` (wildcard index) + `_post_filter_relationships` (entity post-filter for script match)
   - `rel.target == null` → `_wildcard_rel_types` only (pure wildcard, no post-filter needed)
   - Otherwise (Entity or Component target) → `_structural_rel_keys` via `_world._relationship_slot_key(rel)` (exact structural match)
   - Set `_cache_key_valid = false` (relationships now affect cache key)

3. **Modify `without_relationship()`**: Same classification for excludes:
   - `rel._is_query_relationship` → `_post_filter_ex_relationships`
   - `rel.target is Script` → `_post_filter_ex_relationships` (can't exclude structurally for script targets)
   - `rel.target == null` → `_wildcard_ex_rel_types` (exclude all archetypes with that relation type)
   - Otherwise → `_structural_ex_rel_keys` (exact slot key exclusion)

4. **Modify `get_cache_key()`**: Pass structural relationships to QueryCacheKey:

```gdscript
func get_cache_key() -> int:
    if not _cache_key_valid:
        if _world:
            # Filter to structural relationships for cache key
            var structural_rels: Array = []
            for rel in _relationships:
                if not rel._is_query_relationship:
                    structural_rels.append(rel)
            var structural_ex_rels: Array = []
            for rel in _exclude_relationships:
                if not rel._is_query_relationship:
                    structural_ex_rels.append(rel)
            _cache_key = QueryCacheKey.build(
                _all_components, _any_components, _exclude_components,
                structural_rels, structural_ex_rels
            )
            _cache_key_valid = true
        else:
            return -1
    return _cache_key
```

5. **Modify `execute()`**: Change the caching logic. Replace `uses_relationship_filters` with a check for post-filter relationships only:

```gdscript
var has_post_filter_rels := (not _post_filter_relationships.is_empty() or not _post_filter_ex_relationships.is_empty())
```

Cache is valid for structural-only relationship queries (they're part of the cache key now).

6. **Modify `_internal_execute()`**: Pass structural relationship info to `_world._query()`:

```gdscript
var result = _world._query(
    _all_components, _any_components, _exclude_components,
    _enabled_filter, get_cache_key(),
    _structural_rel_keys, _wildcard_rel_types,
    _structural_ex_rel_keys, _wildcard_ex_rel_types
)
```

Replace the existing per-entity relationship filtering block with post-filter-only filtering:

if not _post_filter_relationships.is_empty() or not _post_filter_ex_relationships.is_empty():

11. **Add two helper methods to World**:

```gdscript
## Get archetypes that contain ALL specified relation types (wildcard index intersection)
func _get_archetypes_with_all_relation_types(rel_types: Array) -> Array:
    var result = null
    for rel_path in rel_types:
        var type_archetypes = _relation_type_archetype_index.get(rel_path, {})
        if result == null:
            result = type_archetypes.duplicate()
        else:
            for sig in result.keys():
                if not type_archetypes.has(sig):
                    result.erase(sig)
    return result.values() if result else []

## Check if archetype has any relationship of the specified types
func _archetype_has_any_relation_type(archetype: Archetype, rel_types: Array) -> bool:
    for rel_path in rel_types:
        if _relation_type_archetype_index.has(rel_path):
            if _relation_type_archetype_index[rel_path].has(archetype.signature):
                return true
    return false
```
