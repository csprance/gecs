    ```gdscript
    func test_component_queries_in_relationships():
        # Verify property-query in relation position works as expected post-filter
        # Setup: bob cries in front of someone who eats a lot (value 75), alice cries in front of light eater (value 25)
        var damage_marker = C_IsCryingInFrontOf.new()
        var light_eater = C_Eats.new(25)
        var heavy_eater_1 = C_Eats.new(75)
        var heavy_eater_2 = C_Eats.new(150)

        e_bob.add_relationship(Relationship.new(damage_marker, heavy_eater_1))
        e_alice.add_relationship(Relationship.new(damage_marker, light_eater))
        e_heather.add_relationship(Relationship.new(damage_marker, heavy_eater_2))

        # Property query on target component: find entities whose cry-target eats > 50
        var high_damage_query = Relationship.new(damage_marker, {C_Eats: {"value": {"_gt": 50}}})
        var high_damage_entities = Array(ECS.world.query.with_relationship([high_damage_query]).execute())

        assert_bool(high_damage_entities.has(e_bob)).is_true()    # bob's target eats 75 > 50
        assert_bool(high_damage_entities.has(e_heather)).is_true() # heather's target eats 150 > 50
        assert_bool(high_damage_entities.has(e_alice)).is_false()  # alice's target eats 25, not > 50
        assert_int(high_damage_entities.size()).is_equal(2)

        # Property query on relation position: find entities crying with specific amount
        var relation_query = Relationship.new({C_IsCryingInFrontOf: {"value": {"_gte": 0}}}, null)

        assert_bool(any_crying.has(e_alice)).is_true()

    ```
